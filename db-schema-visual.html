-- ============================================
-- TABLE: properties  (core table)
-- ============================================
CREATE TABLE properties (
  id                UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  ref_code          TEXT UNIQUE NOT NULL,       -- AQ-2026-00041
  owner_id          UUID NOT NULL REFERENCES users(id),
  type_id           INT  NOT NULL REFERENCES property_types(id),
  city_id           INT  NOT NULL REFERENCES cities(id),
  operation         TEXT NOT NULL CHECK (operation IN ('sale', 'rent', 'both')),

  title_ar          TEXT NOT NULL,
  title_fr          TEXT,
  location          GEOGRAPHY(POINT, 4326),   -- PostGIS geolocation

  price             NUMERIC(14,2) NOT NULL,
  price_ai_estimate NUMERIC(14,2),

  status            TEXT DEFAULT 'pending'
                    CHECK (status IN ('active','pending','rejected','sold')),
  image_hash        TEXT[],                   -- للكشف عن التكرار 🤖
  views_count       INT DEFAULT 0,
  created_at        TIMESTAMPTZ DEFAULT NOW()
);

-- Geospatial index (for map queries)
CREATE INDEX idx_properties_location ON properties USING GIST(location);
CREATE INDEX idx_properties_city     ON properties(city_id);
CREATE INDEX idx_properties_price    ON properties(price);
